TRIGGERS:


CREATE TRIGGER trigger1 UPDATE OF customer_sin ON TBLCUSTOMER
BEGIN 
	UPDATE TBLHISTORY SET customer_sin = new.customer_sin WHERE customer_sin = old.customer_sin;
	END


CREATE TRIGGER trigger2 UPDATE OF employees_sin ON TBLEMPLOYEES
BEGIN 
	UPDATE TBLHISTORY SET employees_sin = new.employees_sin WHERE employees_sin = old.employees_sin;
	END
